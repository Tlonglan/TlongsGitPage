{{/* 用法： {{< list-by-taxonomy taxonomy="tags" terms="Hugo,Hextra" match="any" >}} {{< list-by-taxonomy taxonomy="categories" terms="技术,前端" match="all" exclude="旧文" >}} {{< list-by-taxonomy paths="blog/travel,blog/year-end" >}} {{< list-by-taxonomy paths="docs" taxonomy="series" terms="教程" match="any" >}} */}} {{- $taxonomy := .Get "taxonomy" | default "" -}} {{- $termsParam := .Get "terms" | default "" -}} {{- $match := .Get "match" | default "any" -}} {{- $excludeParam := .Get "exclude" | default "" -}} {{- $pathsParam := .Get "paths" | default "" -}} {{- $pages := site.AllPages -}} {{- /* 1. 按路径筛选 (如果提供了 paths) */ -}} {{- if ne $pathsParam "" -}} {{- $pathList := split $pathsParam "," -}} {{- $filtered := slice -}} {{- range $pages -}} {{- $rel := .RelPermalink -}} {{- $matched := false -}} {{- range $pathList -}} {{- $p := trim . " " -}} {{- $prefix := print "/" $p "/" -}} {{- if hasPrefix $rel $prefix -}} {{- $matched = true -}} {{- end -}} {{- end -}} {{- if $matched -}} {{- $filtered = $filtered | append . -}} {{- end -}} {{- end -}} {{- $pages = $filtered -}} {{- end -}} {{- /* 2. 按 taxonomy 筛选 (如果提供了 taxonomy 和 terms) */ -}} {{- if and $taxonomy $termsParam -}} {{- $terms := split $termsParam "," -}} {{- $excludeTerms := slice -}} {{- if $excludeParam -}} {{- $excludeTerms = split $excludeParam "," -}} {{- end -}} {{- $filtered := slice -}} {{- range $pages -}} {{- $pageTax := .Param $taxonomy | default slice -}} {{- $include := false -}} {{- if eq $match "all" -}} {{- $include = true -}} {{- range $terms -}} {{- if not (in $pageTax (trim . " ")) -}} {{- $include = false -}} {{- end -}} {{- end -}} {{- else -}} {{- range $terms -}} {{- if in $pageTax (trim . " ") -}} {{- $include = true -}} {{- end -}} {{- end -}} {{- end -}} {{- if $include -}} {{- if $excludeParam -}} {{- range $excludeTerms -}} {{- if in $pageTax (trim . " ") -}} {{- $include = false -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- if $include -}} {{- $filtered = $filtered | append . -}} {{- end -}} {{- end -}} {{- $pages = $filtered -}} {{- end -}} {{- /* 3. 输出结果 */ -}} {{- if $pages -}}
